Manual Fallback Commands

If buttons don't work, copy and paste these commands into your browser console (F12 > Console):

// Open Projects Dashboard
localStorage.setItem('showProjectDashboard', 'true');
document.querySelector('.project-dashboard')?.style.display = 'flex';
document.querySelector('.project-dashboard')?.style.setProperty('visibility', 'visible', 'important');

// Open Main Menu
document.querySelector('.main-menu-container')?.style.display = 'block';
document.querySelector('.hamburger-button')?.click();

// Find and click New Project button
[...document.querySelectorAll('button')]
  .find(b => b.textContent.includes('New Project'))?.click();
